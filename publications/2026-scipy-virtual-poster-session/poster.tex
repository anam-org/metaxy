% ── COLUMN 1 · THE PROBLEM ────────────────────────────────────
\column{0.33}
\block{\eyebrow{The GPU Economics Problem}\serifhead{One careless rerun blows the weekly GPU budget}}{
  Tabular pipelines forgive a full rerun --- they finish by morning on CPUs.
  Multimodal pipelines do not. A tweak to cropping resolution cascades
  through millions of samples, firing audio, transcription, and embedding
  stages that never touched the frames.

  \vspace{3mm}
  \colorbox{tealsoft}{\parbox{\linewidth-6mm}{\vspace{2mm}%
    \monolabel{The scenario Metaxy handles}\vspace{1mm}\par
    Bump the cropping resolution. Without field-level tracking the full
    pipeline would reprocess --- audio-only stages included. Roughly half
    those stages never touched a frame.\vspace{2mm}}}

  \vspace{3mm}
  \stat{Compute reality}{10--100$\times$ cost}%
       {GPU hours vs.\ CPU hours for the same wall-clock window.}
  \stat{Iteration reality}{Many reruns / day}%
       {Parameter sweeps, model swaps, bugfixes throughout the day.}
  \stat{Decision cost}{Guessing is expensive}%
       {You cannot afford to guess what needs recomputing.}

  \vspace{2mm}
  {\color{ink40}\small
  Orchestrators (Dagster, Airflow), artifact versioning (DVC), online serving
  (Feast), lineage (Hamilton), and versioned datasets (LanceDB, DataChain) all
  stop \emph{above} the field~$\times$~record grain where GPU cost accrues.
  Metaxy fills that gap.}
}

% ── COLUMN 2 · THE IDEA ───────────────────────────────────────
\column{0.34}
\block{\eyebrow{The Key Insight}\serifhead{A version per field, per record}}{
  Each sample carries a \textbf{versioning dictionary} --- one hash per
  field. Each per-field hash composes the upstream field's version with
  the current code version, so a change propagates only along the edges
  that actually consume it.

  \vspace{3mm}
  \begin{center}\svgimg[width=0.86\linewidth]{field_version}\end{center}
  \vspace{1mm}
  \begin{center}\color{ink40}\itshape\large
    Per-field hash = SHA(upstream version, code version).
  \end{center}

  \vspace{3mm}
  {\large
  Field \emph{code versions} mark algorithmic changes; \emph{field
  provenance} folds them with upstream record versions; \emph{record
  provenance} is the sample-level rollup. \texttt{resolve\_update}
  returns the concrete \emph{new}, \emph{stale}, and \emph{orphaned}
  rows.}

  \vspace{3mm}
  \begin{center}\svgimg[width=0.86\linewidth]{field-deps}\end{center}
  \vspace{1mm}
  \begin{center}\color{ink40}\itshape\large
    A change to \texttt{Video.frames} reaches
    \texttt{FaceDetection}, not \texttt{Transcript}.
  \end{center}
}

\block{\eyebrow{Pluggable backends}\serifheadSm{Scale small or large --- as needed}}{
  Metaxy separates \emph{metadata} from storage, orchestration, and
  execution. Version computation is pushed into the metadata store via
  SQL (Ibis); dataframe interop is backend-agnostic via Narwhals.

  \vspace{2mm}
  \stat{Storage}{DuckDB $\to$ warehouse}%
       {DuckDB $\cdot$ ClickHouse $\cdot$ BigQuery $\cdot$ PostgreSQL $\cdot$
        Delta Lake $\cdot$ Iceberg $\cdot$ LanceDB $\cdot$ DuckLake.}
  \stat{Orchestration}{Consume the diff anywhere}%
       {Dagster via \texttt{@metaxify} $\cdot$ Ray for distributed GPU $\cdot$
        any runner over Polars / Arrow.}
